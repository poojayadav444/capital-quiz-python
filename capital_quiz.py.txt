countries = {
    "India": "New Delhi",
    "Russia": "Moscow",
    "United Kingdom": "London",
    "Japan": "Tokyo",
    "France": "Paris",
    "Australia": "Canberra"
}

score = 0

for country, capital in countries.items():
    answer = input(f"What is the capital of {country}? ") # by using f string to insert variables directly into the string using { }

    if answer.strip().lower() == capital.lower():
        print("Correct!")
        score += 1
    else:
        print(f"Wrong! Correct answer is {capital}")

print("\nYour final score is:", score, "/ 6")
